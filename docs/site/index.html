# 1. Install
pip install synthron

# 2. Configure (free keys from aistudio.google.com + console.groq.com)
cp .env.example .env
# Add GEMINI_API_KEY and GROQ_API_KEY to .env

# 3. Run
import asyncio
from synthron import run

result = asyncio.run(run("Your task here"))
print(result)
